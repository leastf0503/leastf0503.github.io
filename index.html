        // Animate eyes for Variant B
        function startEyeAnimation() {
            // Follow mouse movement
            document.addEventListener('mousemove', handleMouseMove);

            // Blink every 3-4 seconds
            function blink() {
                const eye1 = document.getElementById('eye1');
                const eye2 = document.getElementById('eye2');
                
                eye1.classList.add('blinking');
                eye2.classList.add('blinking');
                
                setTimeout(() => {
                    eye1.classList.remove('blinking');
                    eye2.classList.remove('blinking');
                }, 200);
            }

            blinkInterval = setInterval(blink, 3500);
        }

        function handleMouseMove(e) {
            const eye1 = document.getElementById('eye1');
            const eye2 = document.getElementById('eye2');
            
            if (!eye1 || !eye2) return;

            updatePupilPosition(eye1, document.getElementById('pupil1'), e.clientX, e.clientY);
            updatePupilPosition(eye2, document.getElementById('pupil2'), e.clientX, e.clientY);
        }

        function updatePupilPosition(eye, pupil, mouseX, mouseY) {
            const eyeRect = eye.getBoundingClientRect();
            const eyeCenterX = eyeRect.left + eyeRect.width / 2;
            const eyeCenterY = eyeRect.top + eyeRect.height / 2;

            const angle = Math.atan2(mouseY - eyeCenterY, mouseX - eyeCenterX);
            const distance = Math.min(eyeRect.width / 4, 10);

            const pupilX = Math.cos(angle) * distance;
            const pupilY = Math.sin(angle) * distance;

            pupil.style.transform = `translate(calc(-50% + ${pupilX}px), calc(-50% + ${pupilY}px))`;
        }

        // Handle banner interaction
        function handleBannerClick(choice) {
            if (bannerInteracted) return;

            bannerInteracted = true;
            studyData.banner_choice = choice;
            studyData.timestamp_banner_interaction = new Date().toISOString();

            // Calculate time to interaction
            const bannerShown = new Date(studyData.timestamp_banner_shown);
            const bannerInteraction = new Date(studyData.timestamp_banner_interaction);
            studyData.time_to_banner_interaction = (bannerInteraction - bannerShown) / 1000; // in seconds

            console.log(`Banner ${choice}ed after ${studyData.time_to_banner_interaction.toFixed(2)} seconds`);

            // Hide banner
            document.getElementById('cookie-banner').classList.remove('active');
            document.getElementById('modal-overlay').classList.remove('active');

            // Stop eye animation if variant B
            if (bannerVariant === 'B') {
                document.removeEventListener('mousemove', handleMouseMove);
                if (blinkInterval) {
                    clearInterval(blinkInterval);
                }
            }
        }

        // Add to cart
        function addToCart(productName, price) {
            cartCount++;
            document.getElementById('cart-count').textContent = cartCount;

            // Track products before/after banner interaction
            if (!bannerInteracted) {
                studyData.products_before_banner++;
            } else {
                studyData.products_after_banner++;
            }

            studyData.cart_items.push({
                product: productName,
                price: price,
                timestamp: new Date().toISOString(),
                before_banner: !bannerInteracted
            });

            console.log(`Added ${productName} to cart (Total: ${cartCount})`);
        }

        // End study
        function endStudy() {
            if (studyTimer) {
                clearTimeout(studyTimer);
            }

            studyData.timestamp_end = new Date().toISOString();

            // Calculate total time on page
            const start = new Date(studyData.timestamp_start);
            const end = new Date(studyData.timestamp_end);
            studyData.total_time_on_page = (end - start) / 1000; // in seconds

            console.log('Study ended. Final data:', studyData);

            // Hide shop, show summary
            document.getElementById('shop-screen').style.display = 'none';
            document.getElementById('summary-screen').style.display = 'block';

            // Display summary
            displaySummary();
        }

        // Display summary

            const summaryDiv = document.getElementById('summary-data');
            summaryDiv.innerHTML = `

                    ${studyData.participant_id}
                

                

                    Banner-Variante:
                    ${studyData.banner_variant}

                    ${studyData.banner_choice || 'Keine'}
                

                

                    Produkte vor Banner-Interaktion:
                    ${studyData.products_before_banner}
                

                

                    Produkte nach Banner-Interaktion:
                    ${studyData.products_after_banner}
                

                

                    Gesamtzeit auf der Seite:
                    ${studyData.total_time_on_page.toFixed(2)} Sekunden
                

                

                    Gesamt Produkte im Warenkorb:
                    ${cartCount}
                

            `;
        }

        // Download data as JSON
        function downloadData() {
            const dataStr = JSON.stringify(studyData, null, 2);
            const dataBlob = new Blob([dataStr], { type: 'application/json' });
            const url = URL.createObjectURL(dataBlob);
            const link = document.createElement('a');
            link.href = url;
            link.download = `studie-daten-${studyData.participant_id}-${Date.now()}.json`;
            document.body.appendChild(link);
            link.click();
            document.body.removeChild(link);
            URL.revokeObjectURL(url);

            console.log('Data downloaded');
        }
